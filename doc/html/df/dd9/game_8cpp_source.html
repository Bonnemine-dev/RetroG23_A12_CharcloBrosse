52void Game::gameLoop()
53{
54 checkTier();
55 if(running){
56 QElapsedTimer timer;
57 timer.restart();
58 if(itsLoopCounter == 0)itsLoopCounter = NUMBER_LOOP_PER_SECOND;
59 itsEllapsedTime += 0.001;
60
61 if (itsLevel->getItsRemainingEnemies().size() > 0){
62 unsigned short pos = itsLevel->getItsRemainingEnemies().size()-1;
63 if(!(itsLevel->getItsEnemyAppearsTimes().empty()) && itsEllapsedTime >= itsLevel->getItsEnemyAppearsTimes().at(pos))
64 {
65 Enemy * enemy = itsLevel->getItsRemainingEnemies().at(pos);
66 Sides side = itsLevel->getItsEnemyAppearsSides().at(pos);
67 switch(side){
68 case LEFT:
69 itsLevel->getItsSpawnerList().at(0)->appears(enemy);
70 enemy->setItsXSpeed(RIGHT_X);
71 break;
72 case RIGHT:
73 itsLevel->getItsSpawnerList().at(1)->appears(enemy);
74 enemy->setItsXSpeed(LEFT_X);
75 break;
76 }
77
78 itsLevel->appears(enemy);
79 itsEllapsedTime = 0;
80 }
81 }
82 checkAllCollid();
83 moveAll();
84 if(itsLoopCounter % (NUMBER_LOOP_PER_SECOND/FPS) == 0)itsHMI->refreshAll();
85 itsLoopCounter--;
86
87 if(isLevelFinished()){
88 itsScore += itsHMI->getTimerRemainingTime();
89 if (currentLevel != MAX_LEVEL){
90 if (currentTier != checkTier()){
91 currentTier = checkTier();
92 delete itsTileSet;
93 itsTileSet = new TileSet(":/ressources/tileset0.png");
94 }
95 currentLevel++;
96 openLevel();
97 itsHMI->setLevel(itsLevel);
98 itsHMI->displayLevelNumber();
99 itsLoopCounter = NUMBER_LOOP_PER_SECOND;
100 itsEllapsedTime = 0;
101 }
102 else{
103 currentLevel = 1;
104 itsHMI->stopGame();
105 }
106 }
107
108 if(itsPlayer->getItsLivesNb() == 0)
109 {
110 currentLevel = 1;
111 itsHMI->stopGame();
112 }
113 }

693 {
694 itsPlayer->move();
695 }
696 for (Enemy * enemy : itsLevel->getItsEnemiesList()){
697 switch (enemy->getItsType())
698 {
699 case STANDARD:
700 if((itsLoopCounter % (NUMBER_LOOP_PER_SECOND/(STANDARD_ENEMY_SPEED*BLOCK_SIZE))) == 0)
701 {
702 enemy->move();
703 }
